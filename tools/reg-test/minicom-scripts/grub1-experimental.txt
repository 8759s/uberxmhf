# Experimental script to manually feed grub a configuration to boot
# Based upon /use/share/doc/minicom/examples/unixlogin
# See also 'man runscript' and 'man minicom' to understand
#
	print "\nWaiting for grub prompt (Press any key to continue.)"
	expect {
		"Press any key to continue." 
	}
    sleep 2
    send ""

	expect {
		"command-line." 
	}
    sleep 2
    send "c"

	expect {
		"grub>"
	}
    sleep 5
    send "root (hd0,0)"

	expect {
		"grub>"
	}
    sleep 5
    send "a.bc.def.ghij.klmno.pqrstu.vwxyz01"
    send ""

	expect {
		"grub>"
	}
    sleep 5
    send "kernel /init-x86.bin"

	expect {
		"grub>"
	}
    sleep 5
    # \011 is the octal ASCII code for a TAB
    send "module /hyp\011"

	expect {
		"grub>"
	}
    sleep 5
    send "modulenounzip (hd0)+1"

	expect {
		"grub>"
	}
    sleep 5
    send "module /i5_i7\011"
    send ""

	expect {
		"grub>"
	}
    sleep 5
    send "boot"

    exit
